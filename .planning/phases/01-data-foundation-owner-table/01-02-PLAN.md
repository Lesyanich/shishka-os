```typescript
// Group dishes by their category_id, then within each L1, group by L2 subcategory
// For now, group dishes under L2 subcategories where applicable
const groupedDishes = useMemo(() => {
  // Get the L1 category for the current view
  const l1Id = selectedCategory // null means "All"
  const relevantL1Ids = l1Id ? [l1Id] : Array.from(new Set(filtered.map(d => d.category_id).filter(Boolean))) as string[]

  const groups: Array<{ type: 'l2-header'; subcategory: MenuSubcategory; dishCount: number } | { type: 'dish'; dish: MenuDish }> = []

  for (const catId of relevantL1Ids) {
    const l2s = subcategories.get(catId) ?? []
    const catDishes = filtered.filter(d => d.category_id === catId)

    if (l2s.length === 0) {
      // No L2 subcategories — render dishes flat
      for (const dish of catDishes) {
        groups.push({ type: 'dish', dish })
      }
    } else {
      // Has L2 subcategories — group dishes under them
      // For MVP: all dishes under their L1 appear after L2 headers
      // (dishes don't have subcategory_id, so L2 headers are informational)
      for (const l2 of l2s) {
        groups.push({ type: 'l2-header', subcategory: l2, dishCount: 0 })
      }
      for (const dish of catDishes) {
        groups.push({ type: 'dish', dish })
      }
    }
  }
  return groups
}, [filtered, subcategories, selectedCategory])
```
